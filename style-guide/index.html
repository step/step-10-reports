JavaScript Style Guide

1. Spaces between LHS and RHS of an assignment

Good


  let x = 5;

Bad


  let x=5;

2. Appropriate spaces between parts of an expression

Good


  if (x === 5) {
    console.log("x is 5");
  }

Bad


  if(x===5){
    console.log("x is 5");
  }

3. Curly braces on the same line as if

Good


  if (x === 5) {
    console.log("x is 5");
  }

Bad


  if (x === 5)
  {
    console.log("x is 5");
  }

4. Close and open curly brace surrounding an else

Good


  if (x === 5) {
    console.log("x is 5");
  } else {
    console.log("x is not 5");
  }

Bad


  if (x === 5) {
    console.log("x is 5");
  } else
  {
    console.log("x is not 5");
  }

5. Indentation of 4 spaces

Good


  if (x === 5) {
    console.log("x is 5");
  }

Bad


  if (x === 5)
  {
    console.log("x is 5");
  }

6. Semi colons at the end of statements

Good


  let x = 5;
  console.log(x);

Bad


  let x = 5
  console.log(x)

7. Remove unnecessary else, for example max of a number

Good


  let max = x;
  if (y > x) {
    max = y;
  }

Bad


  if (x > y) {
    max = x;
  } else {
    max = y;
  }

8. Single declaration per line

Good


  let x = 5;
  let y = 10;

Bad


  let x = 5, y = 10;

9. Variable names that are longer than a single alphabet

Good


  let customerName = "John Doe";

Bad


  let n = "John Doe";

10. Declaration closer to usage and in the right scope instead of at the beginning of a file

Good


  if (x === 5) {
    let greeting = "Hello";
    console.log(greeting);
  }

Bad


  let greeting;

  if (x === 5) {
    greeting = "Hello";
    console.log(greeting);
  }

11. Providing gaps between blocks of code

Good


  // Block 1
  if (x === 5) {
    console.log("x is 5");
  }

  // Block 2
  console.log("This is another block");

Bad


  if (x === 5) {console.log("x is 5");}console.log("This is another block");
